pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                git url: 'https://github.com/Harshakm37/full_stack_sem.git', branch: 'main'
            }
        }

        stage('Prepare Environment') {
            steps {
                script {
                    // Ensure the target directory exists
                    bat 'mkdir K:\\Java'
                    // Copy the file from the repository
                    bat 'copy /Y Fibo.java K:\\Java'
                }
            }
        }

        stage('Compile') {
            steps {
                script {
                    // Compile the Java file
                    bat '''
                        javac K:\\Java\\Fibo.java -d K:\\Java
                    '''
                }
            }
        }

        stage('Run') {
            steps {
                script {
                    // Run the compiled Java program
                    bat '''
                        java -cp K:\\Java Fibo
                    '''
                }
            }
        }
    }

    post {
        always {
            echo 'Pipeline completed.'
        }
        success {
            echo 'Build succeeded!'
        }
        failure {
            echo 'Build failed.'
        }
    }
}
